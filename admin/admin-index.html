// ── MERCHANTS ───────────────────────────────────────────────── let allMerchantsData = []; async function loadMerchants() { showLoader('Loading merchants...'); try { const res = await fetch(`${API}/admin-merchants`,{headers:{'Authorization':'Bearer '+adminToken}}); const data = await res.json(); hideLoader(); if (!res.ok) { notify('Merchants: '+(data.error||'error'),true); showMerchantEmpty(); return; } allMerchantsData = data.merchants||[]; const payments = allMerchantsData.reduce((s,m)=>s+(m.total_payments||0),0); const volume = allMerchantsData.reduce((s,m)=>s+(m.total_received_kobo||0),0); setText('merch-m-total', data.total||0); setText('merch-m-active', data.active||0); setText('merch-m-payments',payments); setText('merch-m-volume', fmt(volume)); setText('m-merchants', data.total||0); setText('m-merch-volume', fmt(volume)); setText('m-qr-payments', payments); renderMerchantCards(allMerchantsData); renderMerchantTable(allMerchantsData); } catch(e) { hideLoader(); notify('Merchants: '+e.message,true); showMerchantEmpty(); } } function showMerchantEmpty() { document.getElementById('merchant-cards-grid').innerHTML = `
🛒
No merchants yet
Share the portal to onboard traders.
🔗 zillion-mvp.netlify.app/merchant/
Run merchants.sql in Supabase if this is first time.
`; document.getElementById('merchant-table-body').innerHTML = 'No merchants. Run merchants.sql in Supabase first.'; } function filterMerchantTable() { const q = document.getElementById('merch-search')?.value.toLowerCase()||''; const st = document.getElementById('merch-status-filter')?.value||''; const tp = document.getElementById('merch-type-filter')?.value||''; const f = allMerchantsData.filter(m=> (!q || m.business_name?.toLowerCase().includes(q)||m.owner_name?.toLowerCase().includes(q)||m.location?.toLowerCase().includes(q))&& (!st || m.status===st)&&(!tp || m.business_type===tp)); renderMerchantCards(f); renderMerchantTable(f); } function renderMerchantCards(merchants) { const el=document.getElementById('merchant-cards-grid'); if (!merchants.length){el.innerHTML='
No merchants match filter
';return;} const icons={'Food & Vegetables':'🥬','Clothing & Fashion':'👗','Electronics':'📱','Pharmacy & Health':'💊', 'Transport':'🚌','Restaurant & Food':'🍲','Hardware & Tools':'🔧','General Retail':'🏪','Services':'⚙️','Other':'📦'}; el.innerHTML=merchants.map(m=>{ const ini=(m.business_name||'ZL').split(' ').map(w=>w[0]).join('').slice(0,2).toUpperCase(); return `
${ini}
${m.business_name}
${m.merchant_id}
Owner${m.owner_name||'—'}
Type${icons[m.business_type]||'🛒'} ${m.business_type||'—'}
Location${m.location||'—'}
Phone${m.phone||'—'}
Payments${m.total_payments||0}
Volume${fmt(m.total_received_kobo||0)}
Status ${m.status}
Joined${fmtDate(m.registered_at)}
`}).join(''); } function renderMerchantTable(merchants) { document.getElementById('merchant-table-body').innerHTML = merchants.length ? merchants.map(m=>` ${m.merchant_id} ${m.owner_name||'—'} ${m.business_name} ${m.business_type||'—'} ${m.location||'—'} ${m.phone||'—'} ${fmtDate(m.registered_at)} ${m.total_payments||0} ${fmt(m.total_received_kobo||0)} ${m.status} `).join('') : 'No merchants found'; } function suspendMerchant(id, status) { const newSt = status==='ACTIVE'?'SUSPENDED':'ACTIVE'; const sql = `UPDATE merchants SET status='${newSt}' WHERE merchant_id='${id}';`; if (confirm(`Change ${id} to ${newSt}?\n\nRun in Supabase:\n${sql}`)) { alert('Copy this SQL and run in Supabase SQL Editor:\n\n'+sql); } } function exportMerchants() { const rows=[['Merchant ID','Owner','Business','Type','Location','Phone','Status','Registered','Payments','Volume (kobo)']]; allMerchantsData.forEach(m=>rows.push([m.merchant_id,m.owner_name,m.business_name,m.business_type,m.location,m.phone,m.status,m.registered_at,m.total_payments||0,m.total_received_kobo||0])); downloadCSV(rows,'zillion-merchants.csv'); }
